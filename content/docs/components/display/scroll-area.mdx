        <PackageInstall pkg="@ark-ui/react" />

      <DocStep>
        Copy the component source into your app:
        <ComponentSource name="scroll-area" title="components/ui/scroll-area.tsx" />
      </DocStep>
      <DocStep>
        Add the following CSS to your stylesheet (e.g. `styles.css`):

        ```css
        @layer base {
          [data-scope="scroll-area"][data-part="viewport"] {
            scrollbar-width: none;
          }
          [data-scope="scroll-area"][data-part="viewport"]::-webkit-scrollbar {
            display: none;
          }
        }
        ```
      </DocStep>
      <DocStep>
        Update import aliases to match your project setup.
      </DocStep>
    </DocSteps>
  </TabsContent>
</Tabs>

## Usage

```tsx
import * as ScrollArea from "@/components/ui/scroll-area"
```

Read exported parts in `src/components/ui/scroll-area.tsx` and compose the primitive according to the Ark UI pattern for this component.
